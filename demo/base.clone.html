clone 方法可以对一个对象进行克隆，得到一个完全全新的对象（不支持HTMLElement），克隆后的对象属性是存在于prototype的（而不是属于obj自身，JSON.stringify 无法把该属性转换成字符串），可以用点号操作符（如 obj.a ）访问。

var obj1 = { a: 1, b: 2, c: '33'};


;Jx().$package(function(J){
    J.event.on(J.dom.id('extendButton'), 'click', function(evt){
        var obj1 = { a: 1, b: 2, c: '33'};
        var obj2 = { a2: 22, b: '22'};
        var obj3 = { a3: 33, b3: '33'};
        var newObj = J.extend(obj1, obj2, obj3);
        alert('扩展结果: ' + J.json.stringify(newObj));
    });
});
