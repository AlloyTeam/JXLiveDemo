bind 方法可以将一个方法的执行上下文（context）修改为给定对象。 

;Jx().$package(function(J){
    J.event.on(J.dom.id('actionButton'), 'click', function(evt){
        var obj1 = { text: 'aaa'};
        function tips(){
            alert('this.text: ' + this.text);
        }
        tips();//执行时this指向了全局的window对象
        var newTips = J.bind(tips, obj1);
        newTips();//执行时this指向了全局的obj1对象
    });
});
