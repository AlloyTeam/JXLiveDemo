<<点击左边的按钮可以显示或隐藏下面的这段文字

要像对待荷叶上的露珠一样小心翼翼地保护学生幼小的心灵，晶莹透亮的露珠是美丽可爱的，但却十分脆弱，一不小心，就会滚落破碎，不复存在，学生的心灵，如同脆弱的露珠，需要老师的倍加呵护。——苏霍姆林斯基

;Jx().$package(function(J){
    var $toggleButton = J.dom.id('toggleButton');
    var $contentArea = J.dom.id('contentArea');
    J.event.on($toggleButton, 'click', function(evt){
        if(J.dom.isShow($contentArea)){
            J.dom.hide($contentArea);
            $toggleButton.innerHTML = '显示';
        }else{
            J.dom.show($contentArea);
            $toggleButton.innerHTML = '隐藏';
        }
    });
});
